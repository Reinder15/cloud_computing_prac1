STAP 3: MAAK CEPH MONITORS AAN
Op alle 3 nodes:
pveceph mon create

STAP 4: MAAK CEPH MANAGERS AAN
Op alle 3 nodes:
pveceph mgr create

STAP 5: CREËER OSD'S (op alle nodes)
# Check beschikbare disks
lsblk
pvesm status

# Maak partitie met parted zoals hierboven beschreven
parted /dev/sdX  # of de juiste disk
(parted) print free
(parted) mkpart OSD1 <start> <end>
(parted) quit

# Creëer OSD
pveceph osd create /dev/sda4  # gebruik de nieuwe partitie

STAP 6: MAAK CEPH POOL AAN (op node1)
# Voor VM's en containers
pveceph pool create ceph-pool --add_storages

STAP 6B: SETUP CEPHFS (Shared filesystem across nodes)
# Maak Metadata Servers (MDS) aan op alle 3 nodes
# Op node1:
pveceph mds create
# Op node2:
pveceph mds create
# Op node3:
pveceph mds create

# Maak CephFS pools en filesystem (op node1)
ceph osd pool create cephfs_data 32
ceph osd pool create cephfs_metadata 16
ceph fs new cephfs cephfs_metadata cephfs_data

# Voeg toe aan Proxmox storage
pvesm add cephfs cephfs --path /mnt/pve/cephfs --content backup,vztmpl,iso,snippets --nodes node1,node2,node3

# Verify
ceph mds stat
ceph fs ls
ls -la /mnt/pve/cephfs

# Clone git repo naar shared storage
cd /mnt/pve/cephfs
git clone https://github.com/Reinder15/cloud_computing_prac1.git
# Nu toegankelijk op alle nodes via: /mnt/pve/cephfs/cloud_computing_prac1

STAP 7: WORDPRESS IN LXC (KLANT 1 - TRAINING)
https://pve.proxmox.com/pve-docs/pve-admin-guide.html#chapter_pct
# Download Ubuntu/Debian LXC template
pveam update
pveam available | grep ubuntu
pveam download local ubuntu-25.04-standard_25.04-1.1_amd64.tar.zst
pveam list local

# Maak LXC container
pct create 100 local:vztmpl/ubuntu-22.04-standard_22.04-1_amd64.tar.zst \
  --hostname wordpress-training \
  --memory 2048 \
  --net0 name=eth0,bridge=vmbr0,ip=10.24.38.10/24,gw=10.24.38.1 \
  --nameserver 8.8.8.8 \
  --onboot 1 \
  --start 1 \
  --storage ceph-pool

# Start container en installeer WordPress MANUEEL:
pct start 100
pct enter 100

# In de container:
https://developer.wordpress.org/advanced-administration/before-install/howto-install/
